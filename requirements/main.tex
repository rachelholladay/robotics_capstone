\newcommand\CheckTable[1]{%
  \setcounter{question}{0}
  \setcounter{row}{0}
  \resettabtoks
  \loop\ifnum\therow<#1\relax
    \stepcounter{row}
    \addtabtoks{& $\square$ & $\square$ & $\square$ & $\square$ & $\square$ & $\square$ & $\square$ \\}%
  \repeat
  \begin{longtable}{>{\stepcounter{question}Priority}l*{7}{c}}
    \multicolumn{1}{c}{} & Lowest (1) & Low & Medium-Low & Neutral (4) & Medium-High & High & Highest (7) \\
    \printtabtoks
  \end{longtable}%
}
