\section{Risk Management}
\label{sec:risk}
\added[remark={RH, V2}]{We assess risk through the following questions.}
\begin{itemize}
	\item \added[remark={YJ, V2}]{Likelihood: how likely is this failure mode?}
	\item \added[remark={YJ, V2}]{Detectability: how likely can this failure mode be noticed?}
	\item \added[remark={YJ, V2}]{Criticality: how serious are the potential results caused by this failure mode?}
	\item \added[remark={YJ, V2}]{Priority: how much attention should be put in to address this failure mode?} 
\end{itemize}

\added[remark={YJ, V2}]{The scales for priority is assigned based on likelihood, detectability, and criticality. We define the following scales and rank our failure modes accordingly.}\\

\begin{center}
 \begin{tabular}{||c c ||}
 \hline
 Scale & Likelihood  \\ [0.5ex]
 \hline\hline
 1 & Impossible (close to 0\% chance) \\
 \hline
 2 & Unlikely (25\% chance) \\
 \hline
 3 & Possible (50\% chance) \\
 \hline
 4 & Certain (75\% chance) \\
 \hline
 5 & Inevitable (100\% chance) \\
 \hline 
\end{tabular}
\end{center}

\begin{center}
 \begin{tabular}{||c c ||}

 \hline

 \hline

 \hline

 \hline 
\end{tabular}
\end{center}
